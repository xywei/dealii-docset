36 template <int dim, int spacedim>
37 Tensor<1,spacedim>
38 PolarManifold<dim,spacedim>::get_periodicity()
39 {
40  Tensor<1,spacedim> periodicity;
41  // In two dimensions, theta is periodic.
42  // In three dimensions things are a little more complicated, since the only variable
43  // that is truly periodic is phi, while theta should be bounded between
44  // 0 and pi. There is currently no way to enforce this, so here we only fix
45  // periodicity for the last variable, corresponding to theta in 2d and phi in 3d.
46  periodicity[spacedim-1] = 2*numbers::PI;
47  return periodicity;
48 }

50 template <int dim, int spacedim>
51 Point<spacedim>
52 PolarManifold<dim,spacedim>::push_forward(const Point<spacedim> &spherical_point) const
53 {
54  Assert(spherical_point[0] >=0.0,
55  ExcMessage("Negative radius for given point."));
56  const double rho = spherical_point[0];
57  const double theta = spherical_point[1];
58 
59  Point<spacedim> p;
60  if (rho > 1e-10)
61  switch (spacedim)
62  {
63  case 2:
64  p[0] = rho*cos(theta);
65  p[1] = rho*sin(theta);
66  break;
67  case 3:
68  {
69  const double phi= spherical_point[2];
70  p[0] = rho*sin(theta)*cos(phi);
71  p[1] = rho*sin(theta)*sin(phi);
72  p[2] = rho*cos(theta);
73  break;
74  }
75  default:
76  Assert(false, ExcNotImplemented());
77  }
78  return p+center;
79 }

81 template <int dim, int spacedim>
82 Point<spacedim>
83 PolarManifold<dim,spacedim>::pull_back(const Point<spacedim> &space_point) const
84 {
85  const Tensor<1,spacedim> R = space_point-center;
86  const double rho = R.norm();
87 
88  Point<spacedim> p;
89  p[0] = rho;
90 
91  switch (spacedim)
92  {
93  case 2:
94  {
95  p[1] = atan2(R[1],R[0]);
96  if (p[1] < 0)
97  p[1] += 2*numbers::PI;
98  break;
99  }
100 
101  case 3:
102  {
103  const double z = R[2];
104  p[2] = atan2(R[1],R[0]); // phi
105  if (p[2] < 0)
106  p[2] += 2*numbers::PI; // phi is periodic
107  p[1] = atan2(sqrt(R[0]*R[0]+R[1]*R[1]),z); // theta
108  break;
109  }
110 
111  default:
112  Assert(false, ExcNotImplemented());
113  }
114  return p;
115 }

117 template <int dim, int spacedim>
118 DerivativeForm<1,spacedim,spacedim>
119 PolarManifold<dim,spacedim>::push_forward_gradient(const Point<spacedim> &spherical_point) const
120 {
121  Assert(spherical_point[0] >= 0.0,
122  ExcMessage("Negative radius for given point."));
123  const double rho = spherical_point[0];
124  const double theta = spherical_point[1];
125 
126  DerivativeForm<1,spacedim,spacedim> DX;
127  if (rho > 1e-10)
128  switch (spacedim)
129  {
130  case 2:
131  {
132  DX[0][0] = cos(theta);
133  DX[0][1] = -rho*sin(theta);
134  DX[1][0] = sin(theta);
135  DX[1][1] = rho*cos(theta);
136  break;
137  }
138 
139  case 3:
140  {
141  const double phi= spherical_point[2];
142  DX[0][0] = sin(theta)*cos(phi);
143  DX[0][1] = rho*cos(theta)*cos(phi);
144  DX[0][2] = -rho*sin(theta)*sin(phi);
145 
146  DX[1][0] = sin(theta)*sin(phi);
147  DX[1][1] = rho*cos(theta)*sin(phi);
148  DX[1][2] = rho*sin(theta)*cos(phi);
149 
150  DX[2][0] = cos(theta);
151  DX[2][1] = -rho*sin(theta);
152  DX[2][2] = 0;
153  break;
154  }
155 
156  default:
157  Assert(false, ExcNotImplemented());
158  }
159  return DX;
160 }

171 template <int dim, int spacedim>
172 Point<spacedim>
173 SphericalManifold<dim,spacedim>::
174 get_intermediate_point (const Point<spacedim> &p1,
175  const Point<spacedim> &p2,
176  const double w) const
177 {
178  Assert(w >=0.0 && w <= 1.0,
179  ExcMessage("w should be in the range [0.0,1.0]."));
180 
181  const double tol = 1e-10;
182 
183  if ( p1.distance(p2) < tol || w < tol)
184  return p1;
185  else if (w > 1.0 - tol)
186  return p2;
187 
188  // If the points are one dimensional then there is no need for anything but
189  // a linear combination.
190  if (spacedim == 1)
191  return Point<spacedim>(w*p2 + (1-w)*p1);
192 
193  const Tensor<1,spacedim> v1 = p1 - center;
194  const Tensor<1,spacedim> v2 = p2 - center;
195  const double r1 = v1.norm();
196  const double r2 = v2.norm();
197 
198  Assert(r1 > tol && r2 > tol,
199  ExcMessage("p1 and p2 cannot coincide with the center."));
200 
201  const Tensor<1,spacedim> e1 = v1/r1;
202  const Tensor<1,spacedim> e2 = v2/r2;
203 
204  if ((e1 - e2).norm_square() < tol*tol)
205  return Point<spacedim>(center + w*v2 + (1-w)*v1);
206 
207  // Find the angle gamma described by v1 and v2:
208  const double gamma = std::acos(e1*e2);
209 
210  // Find the angle sigma that corresponds to arclength equal to w
211  const double sigma = w * gamma;
212 
213  // Normal to v1 in the plane described by v1,v2,and the origin.
214  // Since p1 and p2 do not coincide n is not zero and well defined.
215  Tensor<1,spacedim> n = v2 - (v2*e1)*e1;
216  Assert( n.norm() > 0,
217  ExcInternalError("n should be different from the null vector."
218  "Probably this means v1==v2 or v2==0."));
219 
220  n /= n.norm();
221 
222  // Find the point Q along O,v1 such that
223  // P1,V,P2 has measure sigma.
224  const Tensor<1,spacedim> P = std::cos(sigma) * e1 + std::sin(sigma) * n;
225 
226  // Project this point on the manifold.
227  return Point<spacedim>(center + (w*r2+(1.0-w)*r1)*P);
228 }

230 template <int dim, int spacedim>
231 Tensor<1,spacedim>
232 SphericalManifold<dim,spacedim>::
233 get_tangent_vector (const Point<spacedim> &p1,
234  const Point<spacedim> &p2) const
235 {
236  Assert(p1 != p2,
237  ExcMessage("p1 and p2 should not concide."));
238 
239  const double r1 = (p1 - center).norm();
240  const double r2 = (p2 - center).norm();
241 
242  const double tolerance = 1e-10;
243 
244  Assert(r1 > tolerance,
245  ExcMessage("p1 cannot coincide with the center."));
246 
247  Assert(r2 > tolerance,
248  ExcMessage("p2 cannot coincide with the center."));
249 
250  const Tensor<1,spacedim> e1 = (p1 - center)/r1;
251  const Tensor<1,spacedim> e2 = (p2 - center)/r2;
252 
253  Assert(e1*e2 + 1.0 > tolerance,
254  ExcMessage("p1 and p2 cannot lie on the same diameter and be opposite "
255  "respect to the center."));
256 
257  // Tangent vector to the unit sphere along the geodesic given by e1 and e2.
258  Tensor<1,spacedim> tg = (e2-(e2*e1)*e1);
259 
260  // There is a special case if e2*e1==1.0, in which case tg==0
261  const double tg_norm = tg.norm();
262  if (tg_norm < tolerance)
263  return p2-p1;
264  else
265  tg /= tg_norm;
266 
267  const double gamma = std::acos(e1*e2);
268 
269  return (r1-r2)*e1 + r1*gamma*tg;
270 }

280 template <int dim, int spacedim>
281 Point<spacedim>
282 SphericalManifold<dim,spacedim>::
283 get_new_point (const std::vector<Point<spacedim> > &vertices,
284  const std::vector<double> &weights) const
285 {
286  const unsigned int n_points = vertices.size();
287 
288  double rho = 0.0;
289  Tensor<1,spacedim> candidate;
290  for (unsigned int i = 0; i<n_points; i++)
291  {
292  const Tensor<1,spacedim> direction (vertices[i]-center);
293  rho += direction.norm()*weights[i];
294  candidate += direction*weights[i];
295  }
296 
297  // Unit norm direction.
298  candidate /= candidate.norm();
299 
300  return center+rho*candidate;
301 }

342 template <int dim, int spacedim>
343 Point<spacedim>
344 CylindricalManifold<dim,spacedim>::
345 get_new_point (const std::vector<Point<spacedim> > &surrounding_points,
346  const std::vector<double> &weights) const
347 {
348  // compute a proposed new point
349  const Point<spacedim> middle = flat_manifold.get_new_point(surrounding_points,
350  weights);
351 
352  double radius = 0;
353  Tensor<1,spacedim> on_plane;
354 
355  for (unsigned int i=0; i<surrounding_points.size(); ++i)
356  {
357  on_plane = surrounding_points[i]-point_on_axis;
358  on_plane = on_plane - (on_plane*direction) * direction;
359  radius += weights[i]*on_plane.norm();
360  }
361 
362  // we then have to project this point out to the given radius from
363  // the axis. to this end, we have to take into account the offset
364  // point_on_axis and the direction of the axis
365  const Tensor<1,spacedim> vector_from_axis = (middle-point_on_axis) -
366  ((middle-point_on_axis) * direction) * direction;
367 
368  // scale it to the desired length and put everything back together,
369  // unless we have a point on the axis
370  if (vector_from_axis.norm() <= tolerance * middle.norm())
371  return middle;
372 
373  else
374  return Point<spacedim>((vector_from_axis / vector_from_axis.norm() * radius +
375  ((middle-point_on_axis) * direction) * direction +
376  point_on_axis));
377 }
