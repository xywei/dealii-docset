62  template <int dim, int spacedim>
63  bool
64  consistency_check (const std::vector< const FiniteElement< dim, spacedim > * > &fes,
65  const std::vector< unsigned int > &multiplicities,
66  const std::vector<std::vector<std::function<const Function<spacedim> *(const typename Triangulation<dim, spacedim>::cell_iterator &) > > > &functions)
67  {
68  AssertThrow(fes.size() > 0,
69  ExcMessage("FEs size should be >=1"));
70  AssertThrow(fes.size() == multiplicities.size(),
71  ExcMessage("FEs and multiplicities should have the same size"));
72 
73  AssertThrow (functions.size() == fes.size() - 1,
74  ExcDimensionMismatch(functions.size(), fes.size()-1));
75 
76  AssertThrow(multiplicities[0] == 1,
77  ExcMessage("First multiplicity should be 1"));
78 
79  const unsigned int n_comp_base = fes[0]->n_components();
80 
81  // start from fe=1 as 0th is always non-enriched FE.
82  for (unsigned int fe=1; fe < fes.size(); fe++)
83  {
84  const FE_Nothing<dim> *fe_nothing = dynamic_cast<const FE_Nothing<dim>*>(fes[fe]);
85  if (fe_nothing)
86  AssertThrow (fe_nothing->is_dominating(),
87  ExcMessage("Only dominating FE_Nothing can be used in FE_Enriched"));
88 
89  AssertThrow (fes[fe]->n_components() == n_comp_base,
90  ExcMessage("All elements must have the same number of components"));
91  }
92  return true;
93  }

104  for (unsigned int fe=1; fe < fes.size(); fe++)

156 template <int dim, int spacedim>
157 FE_Enriched<dim,spacedim>::FE_Enriched (const std::vector< const FiniteElement< dim, spacedim > * > &fes,
158  const std::vector< unsigned int > &multiplicities,
159  const std::vector<std::vector<std::function<const Function<spacedim> *(const typename Triangulation<dim, spacedim>::cell_iterator &) > > > &functions)
160  :
161  FiniteElement<dim,spacedim> (FETools::Compositing::multiply_dof_numbers(fes,multiplicities,false),
162  FETools::Compositing::compute_restriction_is_additive_flags(fes,multiplicities),
163  FETools::Compositing::compute_nonzero_components(fes,multiplicities,false)),
164  enrichments(functions),
165  is_enriched(check_if_enriched(fes)),
166  fe_system(fes,multiplicities)
167 {
168  // descriptive error are thrown within the function.
169  Assert(consistency_check(fes,multiplicities,functions),
170  ExcInternalError());
171 
172  initialize(fes, multiplicities);
173 
174  // resize to be consistent with all FEs used to construct the FE_Enriched,
175  // even though we will never use the 0th element.
176  base_no_mult_local_enriched_dofs.resize(fes.size());
177  for (unsigned int fe=1; fe < fes.size(); fe++)
178  base_no_mult_local_enriched_dofs[fe].resize(multiplicities[fe]);
179 
180  Assert (base_no_mult_local_enriched_dofs.size() == this->n_base_elements(),
181  ExcDimensionMismatch(base_no_mult_local_enriched_dofs.size(),
182  this->n_base_elements()));
183 
184  // build the map: (base_no, base_m) -> vector of local element DoFs
185  for (unsigned int system_index=0; system_index<this->dofs_per_cell;
186  ++system_index)
187  {
188  const unsigned int base_no = this->system_to_base_table[system_index].first.first;
189  if (base_no == 0) // 0th is always non-enriched FE
190  continue;
191 
192  const unsigned int base_m = this->system_to_base_table[system_index].first.second;
193 
194  Assert (base_m < base_no_mult_local_enriched_dofs[base_no].size(),
195  ExcMessage("Size mismatch for base_no_mult_local_enriched_dofs: "
196  "base_index = " + std::to_string(this->system_to_base_table[system_index].second) +
197  "; base_no = " + std::to_string(base_no) +
198  "; base_m = " + std::to_string(base_m) +
199  "; system_index = " + std::to_string(system_index)));
200 
201  Assert (base_m < base_no_mult_local_enriched_dofs[base_no].size(),
202  ExcDimensionMismatch(base_m,
203  base_no_mult_local_enriched_dofs[base_no].size()));
204 
205  base_no_mult_local_enriched_dofs[base_no][base_m].push_back(system_index);
206  }
207 
208  // make sure that local_enriched_dofs.size() is correct, that is equals to DoFs
209  // per cell of the corresponding FE.
210  for (unsigned int base_no = 1; base_no < base_no_mult_local_enriched_dofs.size(); base_no++)
211  {
212  for (unsigned int m=0; m < base_no_mult_local_enriched_dofs[base_no].size(); m++)
213  Assert ( base_no_mult_local_enriched_dofs[base_no][m].size() == fes[base_no]->dofs_per_cell,
214  ExcDimensionMismatch(base_no_mult_local_enriched_dofs[base_no][m].size(),
215  fes[base_no]->dofs_per_cell));
216  }
217 }

255 template <int dim, int spacedim>
256 UpdateFlags
257 FE_Enriched<dim,spacedim>::requires_update_flags (const UpdateFlags flags) const
258 {
259  UpdateFlags out = fe_system.requires_update_flags(flags);
260 
261  if (is_enriched)
262  {
263  // if we ask for values or gradients, then we would need quadrature points
264  if (flags & (update_values | update_gradients))
265  out |= update_quadrature_points;
266 
267  // if need gradients, add update_values due to product rule
268  if (out & update_gradients)
269  out |= update_values;
270  }
271 
272  Assert (!(flags & update_3rd_derivatives),
273  ExcNotImplemented());
274 
275  return out;

279 template <int dim, int spacedim>
280 template <int dim_1>
281 typename FiniteElement<dim,spacedim>::InternalDataBase *
282 FE_Enriched<dim,spacedim>::setup_data (std::unique_ptr<typename FiniteElement<dim,spacedim>::InternalDataBase> fes_data,
283  const UpdateFlags flags,
284  const Quadrature<dim_1> &quadrature) const
285 {
286  Assert ((dynamic_cast<typename FESystem<dim,spacedim>::InternalData *> (fes_data.get()) != nullptr),
287  ExcInternalError());
288  typename FESystem<dim,spacedim>::InternalData *data_fesystem =
289  static_cast<typename FESystem<dim,spacedim>::InternalData *> (fes_data.get());
290 
291  // FESystem::InternalData will be aggregated (owned) by
292  // our InternalData.
293  fes_data.release();
294  InternalData *data = new InternalData(std::unique_ptr<typename FESystem<dim,spacedim>::InternalData>(data_fesystem));
295 
296  // copy update_each from FESystem data:
297  data->update_each = data_fesystem->update_each;
298 
299  // resize cache array according to requested flags
300  data->enrichment.resize(this->n_base_elements());
301 
302  const unsigned int n_q_points = quadrature.size();
303 
304  for (unsigned int base=0; base < this->n_base_elements(); ++base)
305  {
306  data->enrichment[base].resize(this->element_multiplicity(base));
307  for (unsigned int m = 0; m < this->element_multiplicity(base); ++m)
308  {
309  if (flags & update_values)
310  data->enrichment[base][m].values.resize(n_q_points);
311 
312  if (flags & update_gradients)
313  data->enrichment[base][m].gradients.resize(n_q_points);
314 
315  if (flags & update_hessians)
316  data->enrichment[base][m].hessians.resize(n_q_points);
317  }
318  }
319 
320  return data;
321 }

363 template <int dim, int spacedim>
364 void FE_Enriched<dim,spacedim>::initialize (const std::vector<const FiniteElement<dim,spacedim>*> &fes,
365  const std::vector<unsigned int> &multiplicities)
366 {
367  Assert (fes.size() == multiplicities.size(),
368  ExcDimensionMismatch (fes.size(), multiplicities.size()) );
369 
370  // Note that we need to skip every fe with multiplicity 0 in the following block of code
371  this->base_to_block_indices.reinit(0, 0);
372 
373  for (unsigned int i=0; i<fes.size(); i++)
374  if (multiplicities[i]>0)
375  this->base_to_block_indices.push_back( multiplicities[i] );
376 
377  {
378  // If the system is not primitive, these have not been initialized by
379  // FiniteElement
380  this->system_to_component_table.resize(this->dofs_per_cell);
381  this->face_system_to_component_table.resize(this->dofs_per_face);
382 
383  FETools::Compositing::build_cell_tables(this->system_to_base_table,
384  this->system_to_component_table,
385  this->component_to_base_table,
386  *this,
387  false);
388 
389  FETools::Compositing::build_face_tables(this->face_system_to_base_table,
390  this->face_system_to_component_table,
391  *this,
392  false);
393  }
394 
395  // restriction and prolongation matrices are built on demand
396 
397  // now set up the interface constraints for h-refinement.
398  // take them from fe_system:
399  this->interface_constraints = fe_system.interface_constraints;
400 
401  // if we just wrap another FE (i.e. use FE_Nothing as a second FE)
402  // then it makes sense to have support points.
403  // However, functions like interpolate_boundary_values() need all FEs inside
404  // FECollection to be able to provide support points irrespectively whether
405  // this FE sits on the boundary or not. Thus for moment just copy support
406  // points from fe system:
407  {
408  this->unit_support_points = fe_system.unit_support_points;
409  this->unit_face_support_points = fe_system.unit_face_support_points;
410  }
411 
412  // take adjust_quad_dof_index_for_face_orientation_table from FESystem:
413  {
414  this->adjust_line_dof_index_for_line_orientation_table = fe_system.adjust_line_dof_index_for_line_orientation_table;
415  }
416 }

560 template <int dim, int spacedim>
561 template <int dim_1>
562 void
563 FE_Enriched<dim,spacedim>::multiply_by_enrichment
564 (const Quadrature<dim_1> &quadrature,
565  const InternalData &fe_data,
566  const internal::FEValues::MappingRelatedData<dim,spacedim> &mapping_data,
567  const typename Triangulation< dim, spacedim >::cell_iterator &cell,
568  internal::FEValues::FiniteElementRelatedData<dim,spacedim> &output_data) const
569 {
570  // mapping_data will contain quadrature points on the real element.
571  // fe_internal is needed to get update flags
572  // finally, output_data should store all the results we need.
573 
574  // Either dim_1==dim
575  // (fill_fe_values) or dim_1==dim-1
576  // (fill_fe_(sub)face_values)
577  Assert(dim_1==dim || dim_1==dim-1, ExcInternalError());
578  const UpdateFlags flags = fe_data.update_each;
579 
580  const unsigned int n_q_points = quadrature.size();
581 
582  // First, populate output_data object (that shall hold everything requested such
583  // as shape value, gradients, hessians, etc) from each base element.
584  // That is almost identical to FESystem::compute_fill_one_base(),
585  // the difference being that we do it irrespectively of cell_similarity
586  // and use base_fe_data.update_flags
587 
588  // TODO: do we need it only for dim_1 == dim (i.e. fill_fe_values)?
589  if (dim_1 == dim)
590  for (unsigned int base_no = 1; base_no < this->n_base_elements(); base_no++)
591  {
592  const FiniteElement<dim,spacedim> &
593  base_fe = base_element(base_no);
594  typename FiniteElement<dim,spacedim>::InternalDataBase &
595  base_fe_data = fe_data.get_fe_data(base_no);
596  internal::FEValues::FiniteElementRelatedData<dim,spacedim> &
597  base_data = fe_data.get_fe_output_object(base_no);
598 
599  const UpdateFlags base_flags = base_fe_data.update_each;
600 
601  for (unsigned int system_index=0; system_index<this->dofs_per_cell;
602  ++system_index)
603  if (this->system_to_base_table[system_index].first.first == base_no)
604  {
605  const unsigned int
606  base_index = this->system_to_base_table[system_index].second;
607  Assert (base_index<base_fe.dofs_per_cell, ExcInternalError());
608 
609  // now copy. if the shape function is primitive, then there
610  // is only one value to be copied, but for non-primitive
611  // elements, there might be more values to be copied
612  //
613  // so, find out from which index to take this one value, and
614  // to which index to put
615  unsigned int out_index = 0;
616  for (unsigned int i=0; i<system_index; ++i)
617  out_index += this->n_nonzero_components(i);
618  unsigned int in_index = 0;
619  for (unsigned int i=0; i<base_index; ++i)
620  in_index += base_fe.n_nonzero_components(i);
621 
622  // then loop over the number of components to be copied
623  Assert (this->n_nonzero_components(system_index) ==
624  base_fe.n_nonzero_components(base_index),
625  ExcInternalError());
626  for (unsigned int s=0; s<this->n_nonzero_components(system_index); ++s)
627  {
628  if (base_flags & update_values)
629  for (unsigned int q=0; q<n_q_points; ++q)
630  output_data.shape_values[out_index+s][q] =
631  base_data.shape_values(in_index+s,q);
632 
633  if (base_flags & update_gradients)
634  for (unsigned int q=0; q<n_q_points; ++q)
635  output_data.shape_gradients[out_index+s][q] =
636  base_data.shape_gradients[in_index+s][q];
637 
638  if (base_flags & update_hessians)
639  for (unsigned int q=0; q<n_q_points; ++q)
640  output_data.shape_hessians[out_index+s][q] =
641  base_data.shape_hessians[in_index+s][q];
642  }
643  }
644  }
645 
646  Assert (base_no_mult_local_enriched_dofs.size() == fe_data.enrichment.size(),
647  ExcDimensionMismatch(base_no_mult_local_enriched_dofs.size(),
648  fe_data.enrichment.size()));
649  // calculate hessians, gradients and values for each function
650  for (unsigned int base_no = 1; base_no < this->n_base_elements(); base_no++)
651  {
652  Assert (base_no_mult_local_enriched_dofs[base_no].size() == fe_data.enrichment[base_no].size(),
653  ExcDimensionMismatch(base_no_mult_local_enriched_dofs[base_no].size(),
654  fe_data.enrichment[base_no].size()));
655  for (unsigned int m=0; m < base_no_mult_local_enriched_dofs[base_no].size(); m++)
656  {
657  Assert (enrichments[base_no-1][m](cell) != nullptr,
658  ExcMessage("The pointer to the enrichment function is NULL"));
659 
660  Assert (enrichments[base_no-1][m](cell)->n_components == 1,
661  ExcMessage("Only scalar-valued enrichment functions are allowed"));
662 
663  if (flags & update_hessians)
664  {
665  Assert (fe_data.enrichment[base_no][m].hessians.size() == n_q_points,
666  ExcDimensionMismatch(fe_data.enrichment[base_no][m].hessians.size(),
667  n_q_points));
668  for (unsigned int q=0; q<n_q_points; q++)
669  fe_data.enrichment[base_no][m].hessians[q] = enrichments[base_no-1][m](cell)->hessian (mapping_data.quadrature_points[q]);
670  }
671 
672  if (flags & update_gradients)
673  {
674  Assert (fe_data.enrichment[base_no][m].gradients.size() == n_q_points,
675  ExcDimensionMismatch(fe_data.enrichment[base_no][m].gradients.size(),
676  n_q_points));
677  for (unsigned int q=0; q<n_q_points; q++)
678  fe_data.enrichment[base_no][m].gradients[q] = enrichments[base_no-1][m](cell)->gradient(mapping_data.quadrature_points[q]);
679  }
680 
681  if (flags & update_values)
682  {
683  Assert (fe_data.enrichment[base_no][m].values.size() == n_q_points,
684  ExcDimensionMismatch(fe_data.enrichment[base_no][m].values.size(),
685  n_q_points));
686  for (unsigned int q=0; q<n_q_points; q++)
687  fe_data.enrichment[base_no][m].values[q] = enrichments[base_no-1][m](cell)->value (mapping_data.quadrature_points[q]);
688  }
689  }
690  }
691 
692  // Finally, update the standard data stored in output_data
693  // by expanding the product rule for enrichment function.
694  // note that the order if important, namely
695  // output_data.shape_XYZ contains values of standard FEM and we overwrite
696  // it with the updated one in the following order: hessians -> gradients -> values
697  if (flags & update_hessians)
698  {
699  for (unsigned int base_no = 1; base_no < this->n_base_elements(); base_no++)
700  {
701  for (unsigned int m=0; m < base_no_mult_local_enriched_dofs[base_no].size(); m++)
702  for (unsigned int i=0; i < base_no_mult_local_enriched_dofs[base_no][m].size(); i++)
703  {
704  const unsigned int enriched_dof = base_no_mult_local_enriched_dofs[base_no][m][i];
705  for (unsigned int q=0; q<n_q_points; ++q)
706  {
707  const Tensor<2, spacedim> grad_grad = outer_product(output_data.shape_gradients[enriched_dof][q],
708  fe_data.enrichment[base_no][m].gradients[q]);
709  const Tensor<2,spacedim,double> sym_grad_grad = symmetrize (grad_grad) * 2.0; // symmetrize does [s+s^T]/2
710 
711  output_data.shape_hessians [enriched_dof][q]*= fe_data.enrichment[base_no][m].values[q];
712  output_data.shape_hessians [enriched_dof][q]+=
713  sym_grad_grad +
714  output_data.shape_values [enriched_dof][q] * fe_data.enrichment[base_no][m].hessians[q];
715  }
716  }
717  }
718  }
719 
720  if (flags & update_gradients)
721  for (unsigned int base_no = 1; base_no < this->n_base_elements(); base_no++)
722  {
723  for (unsigned int m=0; m < base_no_mult_local_enriched_dofs[base_no].size(); m++)
724  for (unsigned int i=0; i < base_no_mult_local_enriched_dofs[base_no][m].size(); i++)
725  {
726  const unsigned int enriched_dof = base_no_mult_local_enriched_dofs[base_no][m][i];
727  for (unsigned int q=0; q<n_q_points; ++q)
728  {
729  output_data.shape_gradients[enriched_dof][q]*= fe_data.enrichment[base_no][m].values[q];
730  output_data.shape_gradients[enriched_dof][q]+=
731  output_data.shape_values [enriched_dof][q] * fe_data.enrichment[base_no][m].gradients[q];
732  }
733  }
734  }
735 
736  if (flags & update_values)
737  for (unsigned int base_no = 1; base_no < this->n_base_elements(); base_no++)
738  {
739  for (unsigned int m=0; m < base_no_mult_local_enriched_dofs[base_no].size(); m++)
740  for (unsigned int i=0; i < base_no_mult_local_enriched_dofs[base_no][m].size(); i++)
741  {
742  const unsigned int enriched_dof = base_no_mult_local_enriched_dofs[base_no][m][i];
743  for (unsigned int q=0; q<n_q_points; ++q)
744  {
745  output_data.shape_values[enriched_dof][q] *= fe_data.enrichment[base_no][m].values[q];
746  }
747  }
748  }
749 }

868 FE_Enriched<dim,spacedim>::
869 compare_for_face_domination (const FiniteElement<dim,spacedim> &fe_other) const
870 {
871  // need to decide which element constrain another.
872  // for example Q(2) dominate Q(4) and thus some DoFs of Q(4) will be constrained.
873  // If we have Q(2) and Q(4)+POU, then it's clear that Q(2) dominates,
874  // namely our DoFs will be constrained to make field continuous.
875  // However, we need to check for situations like Q(4) vs Q(2)+POU.
876  // In that case the domination for the underlying FEs should be the other way,
877  // but this implies that we can't constrain POU dofs to make the field continuous.
878  // In that case, through an error
879 
880  // if it's also enriched, do domination based on each one's FESystem
881  if (const FE_Enriched<dim,spacedim> *fe_enr_other
882  = dynamic_cast<const FE_Enriched<dim,spacedim>*>(&fe_other))
883  {
884  return fe_system.compare_for_face_domination(fe_enr_other->get_fe_system());
